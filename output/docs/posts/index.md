<script>
(function () {
  const cards = Array.from(document.querySelectorAll("[data-blog-card]"));
  if (!cards.length) return;

  const tagFilter = document.getElementById("tag-filter");
  const authorFilter = document.getElementById("author-filter");
  const searchInput = document.getElementById("post-search");

  const unique = (values) =>
    Array.from(new Set(values.filter(Boolean))).sort((a, b) => a.localeCompare(b));

  const tags = unique(
    cards.flatMap((card) => card.dataset.tags?.split(",") ?? [])
  );
  const authors = unique(
    cards.flatMap((card) => card.dataset.authors?.split(",") ?? [])
  );

  const renderChips = (collection, target) => {
    target.innerHTML = "";
    collection.forEach((value) => {
      const chip = document.createElement("button");
      chip.type = "button";
      chip.textContent = value;
      chip.dataset.value = value;
      chip.className = "filter-chip";
      chip.addEventListener("click", () => {
        chip.classList.toggle("active");
        applyFilters();
      });
      target.appendChild(chip);
    });
  };

  renderChips(tags, tagFilter);
  renderChips(authors, authorFilter);

  const activeValues = (target) =>
    Array.from(target.querySelectorAll(".filter-chip.active")).map(
      (el) => el.dataset.value
    );

  const applyFilters = () => {
    const query = searchInput.value.trim().toLowerCase();
    const selectedTags = activeValues(tagFilter);
    const selectedAuthors = activeValues(authorFilter);

    cards.forEach((card) => {
      const title = card.dataset.title ?? "";
      const text = (card.querySelector(".blog-card__summary")?.textContent ?? "").toLowerCase();
      const normalize = (datasetValue) =>
        (datasetValue ?? "")
          .split(",")
          .map((item) => item.trim())
          .filter(Boolean);

      const cardTags = normalize(card.dataset.tags);
      const cardAuthors = normalize(card.dataset.authors);

      const matchesQuery =
        !query ||
        title.includes(query) ||
        text.includes(query);

      const matchesTags =
        !selectedTags.length ||
        selectedTags.every((tag) => cardTags.includes(tag));

      const matchesAuthors =
        !selectedAuthors.length ||
        selectedAuthors.some((author) => cardAuthors.includes(author));

      card.hidden = !(matchesQuery && matchesTags && matchesAuthors);
    });
  };

  searchInput.addEventListener("input", applyFilters);
})();
</script>
